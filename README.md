###python的猜数字小游戏
```import random
def caishu():
    i = 0
    key=random.randint(1,20)
    while i < 5 :
        guss=int(input("请输入一个你猜的数:"))
        if guss == key :
            print("恭喜你你猜对了,你好牛")
            break
        elif guss > key:
            print("guss > key, 你猜的数大了，就这？？")
        else:
            print("guss < key,你猜的数小了，就这？？")
        i=i+1
    else:
        print("就这就这就这？？/lj再见")
        print("答案是%s"%key)
caishu()
```
